from flask import Flask, request, jsonify
import nltk
from nltk.chat.util import Chat, reflections

app = Flask(__name__)

# Sample pairs for the chatbot
pairs = [
    [
        r"my name is (.*)",
        ["Hello %1, how can I assist you today?"]
    ],
    [
        r"what is the admission process?",
        ["The admission process involves filling out an application form, submitting required documents, and attending counseling sessions."]
    ],
    [
        r"what are the eligibility criteria?",
        ["Eligibility criteria vary by course. Generally, you need to have completed your 10+2 with relevant subjects."]
    ],
    [
        r"what about scholarships?",
        ["There are various scholarships available based on merit and financial need. You can check the official website for more details."]
    ],
    [
        r"quit",
        ["Thank you for your inquiry. Have a great day!"]
    ],
]

chatbot = Chat(pairs, reflections)

@app.route("/chat", methods=["POST"])
def chat():
    user_input = request.json.get("message")
    response = chatbot.respond(user_input)
    return jsonify({"response": response})

if __name__ == "__main__":
    app.run(debug=True)
